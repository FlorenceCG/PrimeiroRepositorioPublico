enquanto i <= n faca
         enquanto j <= i faca
                  se((i mod j) = 0) entao
                     cont <- cont + 1
                  fimse

         j <- j + 1
         fimenquanto

         se (cont = 2) então
            escreval

            s <- s + i

         fimse
